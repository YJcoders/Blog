# 1、假设第一个值为最大值，遍历，每个元素与之比较

let arr = [4, 53, 53, 2, 7, 1]
function arrMax(arr) {
  let max = arr[0]
  for (let i = 1;i < arr.length;i++) {
    if (arr[i] > max) {
      let temp = max
      max = arr[i]
      arr[i] = temp
    }
  }
  return max
}
console.log(arrMax(arr));
